## CI门禁配置

### GitHub Actions配置

```yaml
# .github/workflows/vibe-coding.yml
name: Vibe Coding Audit

on: [push, pull_request]

jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      
      - name: 检查铁律1：禁止批量生成
        run: |
          # 检查单次提交文件数量
          if [ $(git diff --name-only HEAD~1 HEAD | wc -l) -gt 5 ]; then
            echo "❌ 单次提交文件数量过多，请单模块开发"
            exit 1
          fi
      
      - name: 检查铁律2：禁止修改锁档
        run: |
          # 检查是否修改锁档文件
          if git diff --name-only HEAD~1 HEAD | grep -q "architecture\|dependencies"; then
            echo "❌ 检测到修改锁档文件，请回到阶段2重新定稿"
            exit 1
          fi
      
      - name: 检查铁律3：禁止编造接口
        run: |
          # 检查API调用是否存在于接口定义中
          # 自定义检查逻辑
      
      - name: 检查铁律6：空项目必须成功
        run: |
          # 检查项目是否能成功启动
          npm install && npm run build
      
      - name: 检查测试覆盖率
        run: |
          # 检查测试覆盖率是否≥80%
          npm run test:coverage
          if [ $(cat coverage/coverage-summary.json | jq '.total.lines.pct') -lt 80 ]; then
            echo "❌ 测试覆盖率不足80%"
            exit 1
          fi
      
      - name: 检查代码风格
        run: |
          # 检查lint是否通过
          npm run lint
```
